class Car:
    def __init__(self, make, model, year, color):
        self.make = make
        self.model = model
        self.year = year
        self.color = color

    def start_engine(self):
        return f"The {self.make} {self.model}'s engine has started."

    def car_info(self):
        return f"{self.year} {self.color} {self.make} {self.model}"
